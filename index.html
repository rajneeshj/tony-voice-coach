function speakLikeTony(text) { if ('speechSynthesis' in window) { // Cancel any ongoing speech before starting new speech window.speechSynthesis.cancel(); const utterance = new SpeechSynthesisUtterance(text); // Elevate energy and natural human cadence utterance.rate = 1.15; utterance.pitch = 0.95; utterance.lang = 'en-US'; // Find the most premium, natural sounding human voice available on your device const voices = window.speechSynthesis.getVoices(); // Priority list for high-quality natural voices const premiumVoice = voices.find(voice => voice.lang.startsWith('en') && (voice.name.includes('Natural') || voice.name.includes('Google') || voice.name.includes('David') || voice.name.includes('Zira')) ); if (premiumVoice) { utterance.voice = premiumVoice; } else { // Fallback to standard US English if premium isn't found const standardVoice = voices.find(voice => voice.lang === 'en-US'); if (standardVoice) utterance.voice = standardVoice; } utterance.onend = () => { resetButton(); }; utterance.onerror = () => { resetButton(); }; window.speechSynthesis.speak(utterance); } else { resetButton(); } }
